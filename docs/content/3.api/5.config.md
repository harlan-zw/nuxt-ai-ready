```ts [nuxt.config.ts]
export default defineNuxtConfig({
  aiReady: {
    // options
  }
})
```

## `enabled`

- **Type:** `boolean`{lang="ts"}
- **Default:** `true`{lang="ts"}

```ts
aiReady: {
  enabled: process.env.NODE_ENV === 'production'
}
```

## `debug`

- **Type:** `boolean`{lang="ts"}
- **Default:** `false`{lang="ts"}

Enable debug logging for module operations.

## `mdreamOptions`

- **Type:** `HTMLToMarkdownOptions & { preset?: 'minimal' }`{lang="ts"}
- **Default:** `{ preset: 'minimal' }`{lang="ts"}

Configure [mdream](https://github.com/harlan-zw/mdream) HTML-to-markdown conversion.

```ts
export default defineNuxtConfig({
  aiReady: {
    mdreamOptions: {
      preset: 'minimal'
    }
  }
})
```

## `markdownCacheHeaders`

- **Type:** `{ maxAge?: number, swr?: boolean }`{lang="ts"}
- **Default:** `{ maxAge: 3600, swr: true }`{lang="ts"}

Cache settings for runtime markdown endpoints.

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `maxAge` | `number` | `3600` | Cache duration in seconds |
| `swr` | `boolean` | `true` | Stale-while-revalidate |

## `llmsTxt`

- **Type:** `LlmsTxtConfig`{lang="ts"}
- **Default:** Auto-generated

Configure llms.txt generation.

```ts
export default defineNuxtConfig({
  aiReady: {
    llmsTxt: {
      sections: [
        {
          title: 'API Reference',
          links: [
            { title: 'REST API', href: '/docs/api', description: 'API docs' }
          ]
        }
      ],
      notes: 'Built with Nuxt AI Ready'
    }
  }
})
```

**LlmsTxtSection:**

| Property | Type | Description |
|----------|------|-------------|
| `title` | `string` | Section title |
| `description` | `string \| string[]` | Section description |
| `links` | `{ title, href, description? }[]` | Links in section |

**LlmsTxtConfig:**

| Property | Type | Description |
|----------|------|-------------|
| `sections` | `LlmsTxtSection[]` | Custom sections |
| `notes` | `string \| string[]` | Notes at end |

## `contentSignal`

- **Type:** `false | { aiTrain?: boolean, search?: boolean, aiInput?: boolean }`{lang="ts"}
- **Default:** `false`{lang="ts"}

Content Signal directives for robots.txt. See [Content Signals guide](/docs/ai-ready/guides/content-signals).

```ts
export default defineNuxtConfig({
  aiReady: {
    contentSignal: {
      aiTrain: false, // Block training
      search: true, // Allow search indexing
      aiInput: true, // Allow RAG/grounding
    }
  }
})
```

## `mcp`

- **Type:** `{ tools?: boolean, resources?: boolean }`{lang="ts"}
- **Default:** `{ tools: true, resources: true }`{lang="ts"}

Control MCP features when `@nuxtjs/mcp-toolkit` installed.

```ts
export default defineNuxtConfig({
  aiReady: {
    mcp: {
      tools: true, // list_pages, search_pages_fuzzy
      resources: true, // pages resource
    }
  }
})
```

See [MCP guide](/docs/ai-ready/guides/mcp) for tool/resource details.

## `cacheMaxAgeSeconds`

- **Type:** `number`{lang="ts"}
- **Default:** `600`{lang="ts"} (10 minutes)

Cache duration for llms.txt route handlers. Uses stale-while-revalidate.

```ts
export default defineNuxtConfig({
  aiReady: {
    cacheMaxAgeSeconds: 3600 // 1 hour
  }
})
```

## `runtimeIndexing`

- **Type:** `{ enabled?: boolean, storage?: string, ttl?: number }`{lang="ts"}
- **Default:** `undefined`{lang="ts"} (disabled)

Enable runtime page indexing. Pages are indexed on-demand as they're visited, without requiring prerendering.

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `enabled` | `boolean` | `false` | Enable runtime indexing |
| `storage` | `string` | `'ai-ready'` | Unstorage mount point prefix |
| `ttl` | `number` | `0` | Re-index after N seconds (0 = never) |

```ts
export default defineNuxtConfig({
  aiReady: {
    runtimeIndexing: {
      enabled: true,
      storage: 'ai-ready',
      ttl: 3600 // re-index after 1 hour
    }
  },
  // Production: configure persistent storage
  nitro: {
    storage: {
      'ai-ready': { driver: 'redis', url: process.env.REDIS_URL }
    }
  }
})
```

**How it works:**

1. On server start, prerendered pages are loaded into storage
2. As pages are visited, they're indexed in the background via `waitUntil`
3. MCP tools and llms.txt read from the unified storage

**Storage drivers for production:**

```ts
// Cloudflare KV
nitro: {
  storage: {
    'ai-ready': { driver: 'cloudflare-kv-binding', binding: 'AI_READY_KV' }
  }
}

// Vercel KV
nitro: {
  storage: {
    'ai-ready': { driver: 'vercel-kv' }
  }
}
```

See [Nitro Storage](https://nitro.build/guide/storage) for all available drivers.
